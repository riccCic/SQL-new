CREATE VIEW MEAL_ORDER_WITH_DETAILS AS 
    SELECT MO.*, M.PRICE, M.CALORIES FROM MEAL_ORDER MO, MEAL M 
    WHERE MO.MEAL_NAME = M.NAME;
